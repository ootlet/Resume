Details

SimpleTech Task Manager is a Windows Forms application designed to seamlessly assign, track, and manage technician tasks across multiple shifts. It features a user-friendly interface for task allocation, acknowledgment, and completion tracking, ensuring operational efficiency and accountability.

Workflow

  1. Administrators can log in to assign tasks to technicians for a specific shift.
  2. Technicians log in to view, acknowledge, and sign off on their assigned tasks.
  3. Tasks can be updated in real-time to reflect their current status.
  4. Completion of tasks is recorded for audit purposes, ensuring seamless multi-shift operations.

Code


// Task Assignment Logic
public void AssignTask(string technician, string taskDetails) {
    Task newTask = new Task {
        TechnicianName = technician,
        Details = taskDetails,
        Status = "Assigned",
        AssignedTime = DateTime.Now
    };
    taskList.Add(newTask);
    UpdateTaskView();
}

// Task Completion Logic
public void CompleteTask(int taskId) {
    var task = taskList.FirstOrDefault(t => t.Id == taskId);
    if (task != null) {
        task.Status = "Completed";
        task.CompletedTime = DateTime.Now;
        UpdateTaskView();
    }
}

// Update UI with Current Tasks
private void UpdateTaskView() {
    taskDataGrid.DataSource = taskList.ToList();
    taskDataGrid.Refresh();
}
